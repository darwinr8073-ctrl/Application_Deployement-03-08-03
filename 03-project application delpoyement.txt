✅ STEP 7 — Create Production Dockerfile
nano Dockerfile
FROM node:18 AS build
WORKDIR /app
COPY package*.json ./
RUN npm install
COPY . .
RUN npm run build

FROM nginx:alpine
COPY --from=build /app/build /usr/share/nginx/html
EXPOSE 80
CMD ["nginx", "-g", "daemon off;"]

✅ STEP 15 — Jenkins Job (CI/CD)

*/dev
*/main

Trigger:

Poll SCM → H/2 * * * *
Build Script (MERGE LOGIC)
#!/bin/bash

REPO_DEV=yourname/react-dev
REPO_PROD=yourname/react-prod

BRANCH=$(git rev-parse --abbrev-ref HEAD)

docker build -t react-app .

if [ "$BRANCH" = "dev" ]; then
 docker tag react-app $REPO_DEV
 docker push $REPO_DEV
fi

if [ "$BRANCH" = "main" ]; then
 docker tag react-app $REPO_PROD
 docker push $REPO_PROD

 docker stop react-app || true
 docker rm react-app || true
 docker run -d -p 80:80 --name react-app $REPO_PROD
fi
